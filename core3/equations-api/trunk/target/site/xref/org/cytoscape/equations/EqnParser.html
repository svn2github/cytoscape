33  import java.util.Map;
34  import java.util.Set;
35  import org.cytoscape.equations.parse_tree.*;

38  public interface EqnParser {
39  	/***
40  	 *  After registering an attribute function "func" it can be used in attribute equations.
41  	 *  @param func  the function that will be registered
42  	 *  @throws IllegalArgumentException will be thrown if "func" is null or the function has previously been registered
43  	 */
44  	public void registerFunction(final Function func) throws IllegalArgumentException;
45  
46  	/***
47  	 *  @return the function associated with the name "functionName" or null if no such function exists
48  	 */
49  	public Function getFunction(final String functionName);
50  
51  	/***
52  	 *  @return the set of currently registered functions
53  	 */
54  	public Set<Function> getRegisteredFunctions();
55  
56  	/***
57  	 *  @param eqn                  a valid attribute equation which must start with an equal sign
58  	 *  @param attribNameToTypeMap  a list of existing attribute names and their types
59  	 *  @return true if the parse succeeded otherwise false
60  	 */
61  	public boolean parse(final String eqn, final Map<String, Class> attribNameToTypeMap);
62  
63  	/***
64  	 *  @return the result type of the parsed equation if the parse succeeded, otherwise null
65  	 */
66  	public Class getType();
67  
68  	/***
69  	 *  If parse() failed, this will return the last error messages.
70  	 *  @return the last error message of null
71  	 */
72  	public String getErrorMsg();
73  
74  	/***
75  	 *  @return all the variable names that have been detected in the most recently parsed equation
76  	 */
77  	public Set<String> getVariableReferences();
78  
79  	/***
80  	 *  @return the parse tree.  Must only be called if parse() returns true!
81  	 */
82  	public Node getParseTree();
83  }
