30  package org.cytoscape.equations;
31  
32  
33  import java.util.Map;
34  
35  
36  /*** Compiler that compiles equations to byte (non-Java) code. */
37  public interface EqnCompiler {
38  	/*** Compiles a string to byte code plus some metadata as encapsulated by the <code>Equation</code> class
39  	 *  @param equation               a textual representation of an equation
40  	 *  @param variableNameToTypeMap  a mapping from variable names occuring in "equation" to their respective types
41  	 */
42  	boolean compile(final String equation, final Map<String, Class> variableNameToTypeMap);
43  
44  	/*** Returns an error message for an error encountered during a call to compile, if any.
45  	 *  @return a human readable error message
46  	 *  Please note that it is probably a good idea to only call this method if compile() returned false.
47  	 */
48  	String getLastErrorMsg();
49  
50  	/*** If compile() returns true, this returns the result of the compilation.
51  	 *  @return an <code>Equation</code> representing the code compiled from the string passed into compile()
52  	 */
53  	Equation getEquation();
54  
55  	/***
56  	 *  A factory method that returns an Equation that always fails at runtime.  The reason for
57  	 *  why this exists is that the returned error equations can subsitute for real equations as
58  	 *  dummies.
59  	 *
60  	 *  @param equation      an arbitrary string that is usually a syntactically invalid equation
61  	 *  @param type          the return type of the error equation
62  	 *  @param errorMessage  the runtime error message that the returned equation will produce
63  	 *
64  	 *  @return the error equation
65  	 */
66  	public Equation getErrorEquation(final String equation, final Class type, final String errorMessage);
67  }
